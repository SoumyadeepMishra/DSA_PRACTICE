# Minimum Deletions to Make Character Frequencies Unique

```cpp

class Solution {
public:
    int minDeletions(string s) {
        int n = s.length();
        vector<int> freq(26, 0);

        for(char &ch : s) {
            int idx = ch - 'a';
            freq[idx]++;
        }

        unordered_set<int> st;
        int result = 0;

        for(int &f : freq) {

            while(st.find(f) != st.end()) {
                f--;
                result++;
            }
            if(f > 0)
                st.insert(f);
        }
        return result;
    }
};
```
