# Minimum Rounds to Complete All Tasks

```cpp

class Solution {
public:
    int minimumRounds(vector<int>& tasks) {
        int n = tasks.size();

        // map to store the frequency
        unordered_map<int, int> mp;
        for(int &task : tasks) {
            mp[task]++;
        }
        
        int round = 0;

        // iterate in the map
        for(auto &it : mp) {
            int count = it.second;

            // main condition
            if(count == 1) // only single element
                return -1;

            if(count % 3 == 0) { // 3*k
                round += (count/3);
            }
            else {
                round += (count/3 + 1);
            }
        }
        return round;
    }
};
```
