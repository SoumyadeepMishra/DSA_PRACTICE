# Minimum Equal Sum of Two Arrays After Replacing Zeros

```cpp

class Solution {
public:

    long long totalSum(vector<int> &nums) {
        long long sum = 0;

        for(auto &num : nums) {
            if(num == 0)
                sum += 1;
            sum += num;
        }

        return sum;
    }

    long long zeroCount(vector<int> &nums) {
        long long zero = 0;

        for(auto &num : nums) {
            if(num == 0)
                zero++;
        }

        return zero;
    }

    long long minSum(vector<int>& nums1, vector<int>& nums2) {
        // count number of zero in both array
        long long zero1 = zeroCount(nums1);
        long long zero2 = zeroCount(nums2);

        // 0 ke 1 diye replace kore tarpor sum ber koro. karon 1 min positive integer
        long long sum1 = totalSum(nums1);
        long long sum2 = totalSum(nums2);

        // main condition
        if(sum1 < sum2 && zero1 == 0)
            return -1;

        if(sum2 < sum1 && zero2 == 0)
            return -1;

        return max(sum1, sum2);
    }
};
```
