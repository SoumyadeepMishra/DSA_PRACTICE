# Least Number of Unique Integers after K Removals

```cpp

class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        int n = arr.size();

        // store frequency in a map
        unordered_map<int, int> mp;
        for(auto &x : arr) {
            mp[x]++;
        }

        vector<int> freq;
        for(auto &it : mp) {
            int f = it.second;
            freq.push_back(f);
        }

        sort(freq.begin(), freq.end());

        for(int i=0;i<freq.size();i++) {

            k = k - freq[i];
            if(k < 0) { // no more to delete
                return freq.size() - i;
            }

        }

        return 0; // all elements deleted
    }
};
```
