#  Break a Palindrome

```cpp

class Solution {
public:
    string breakPalindrome(string palindrome) {
        int n = palindrome.size();

        if(n == 1)
            return "";

        // just check for n/2 characters
        for(int i=0;i<n/2;i++) {
            if(palindrome[i] != 'a') {
                palindrome[i] = 'a';
                return palindrome;
            }
        }

        // all are a's
        palindrome[n-1] = 'b';
        return palindrome;
    }
};

```
