# Maximum Number of Integers to Choose From a Range I

```cpp

class Solution {
public:
    int maxCount(vector<int>& banned, int n, int maxSum) {
        int size = banned.size();

        // set e rakhbo banned element gulo ke 
        unordered_set<int> st(banned.begin(), banned.end());

        int maxNumber = 0;
        int sum = 0;

        for(int i=1;i<=n;i++) {

           if(!st.count(i)){
             if(sum + i > maxSum)
                return maxNumber;
            sum += i;
            maxNumber++;
           }

        }

        return maxNumber;
    }
};
```
