# Bag of Tokens

```cpp

class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int power) {
        int n = tokens.size();
        int score = 0;

        // sort the array
        sort(tokens.begin(), tokens.end());

        int i = 0;
        int j = n-1;
        int maxScore = 0;

        while(i <= j) {
            if(power >= tokens[i]) {
                power -= tokens[i];
                score++;
                maxScore = max(maxScore, score);
                i++;
            }
            else if(score >= 1) {
                score -= 1;
                power += tokens[j];
                maxScore = max(maxScore, score);
                j--;
            }
            else {
                return maxScore;
            }
        }

        return maxScore;
    }
};
```
