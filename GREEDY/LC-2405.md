# Optimal Partition of String

```cpp

class Solution {
public:
    int partitionString(string s) {
        int n = s.length();

        // lastSeen vector
        vector<int> lastSeen(26, -1);
        int currStart = 0; // start of current substring
        int result = 0;

        for(int i=0;i<n;i++) {
            char ch = s[i];

            // main condition
            if(lastSeen[ch-'a'] >= currStart) {
                result++;
                currStart = i;
            }
            lastSeen[ch-'a'] = i;
        }
        return result+1;
    }
};

```
