# Eliminate Maximum Number of Monsters

```cpp

class Solution {
public:
    int eliminateMaximum(vector<int>& dist, vector<int>& speed) {
        int n = dist.size();

        // create time array
        vector<int> time(n);
        for(int i=0;i<n;i++) {
            int t = ceil((float)dist[i]/speed[i]);
            time[i] = t;
        }

        sort(time.begin(), time.end());

        int count = 1; // killed the first monster
        int gunChargeTime = 1; // after 1st kill gun is charged

        for(int i=1;i<n;i++) {

            if(time[i] - gunChargeTime <= 0)
                return count;

            count += 1;
            gunChargeTime += 1;

        }

        return count;

    }
};
```
