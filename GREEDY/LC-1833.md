# Maximum Ice Cream Bars

```cpp

class Solution {
public:
    int maxIceCream(vector<int>& costs, int coins) {
        // first sort the vector in ascending order
        sort(costs.begin(), costs.end());

        int result = 0;

        // traverse
        for(auto &cost : costs) {
            if(coins >= cost) {
                result++;
                coins -= cost;
            }
            else
                return result;
        }
        return result;
    }
};
```
