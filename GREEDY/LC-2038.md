# Remove Colored Pieces if Both Neighbors are the Same Color

```cpp

class Solution {
public:
    bool winnerOfGame(string colors) {
        int n = colors.length();

        int aliceSteps = 0;
        int bobSteps = 0;

        for(int i=1;i<n-1;i++) {
            // main condition
            if(colors[i-1] == colors[i] && colors[i] == colors[i+1]) {

                if(colors[i] == 'A')
                    aliceSteps++;

                else
                    bobSteps++;
            }
              
        }
        
        return aliceSteps > bobSteps ? true : false;
    }
};
```
