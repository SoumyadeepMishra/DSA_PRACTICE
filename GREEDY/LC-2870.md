# Minimum Number of Operations to Make Array Empty

```cpp

class Solution {
public:
    int minOperations(vector<int>& nums) {
        unordered_map<int, int> mp;

        for(auto &num : nums) {
            mp[num]++;
        }

        int result = 0;

        // iterate through the map
        for(auto &it : mp) {
            int freq = it.second;

            // freq == 1 then no answer possible
            if(freq == 1)   
                return -1;

            result += ceil((double)freq/3);
        }

        return result;
    }
};
```
