#  Earliest Possible Day of Full Bloom

```cpp

class Solution {
public:
    int earliestFullBloom(vector<int>& plantTime, vector<int>& growTime) {
        int n = plantTime.size();

        // store as a pair to sort in descending order in grow time
        vector<pair<int, int>> v(n);

        for(int i=0;i<n;i++) {
            v[i] = {plantTime[i], growTime[i]};
        }

        auto lambda = [](pair<int, int> &p1, pair<int, int> &p2) {
            return p1.second > p2.second;
        };

        // sort the vector
        sort(v.begin(), v.end(), lambda);

        int maxBloomDays = 0;
        int prevPlantTime = 0;

        for(int i=0;i<n;i++) {
            int currPlantTime = v[i].first;
            int currGrowTime = v[i].second;
            prevPlantTime += currPlantTime;

            int plantBloomTime = prevPlantTime + currGrowTime;
            maxBloomDays = max(maxBloomDays, plantBloomTime);
        }

        return maxBloomDays;
    }
};
```
