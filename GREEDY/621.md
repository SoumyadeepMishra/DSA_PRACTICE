# Task Scheduler

```cpp

class Solution {
public:
    int leastInterval(vector<char>& tasks, int n) {
        int size = tasks.size();

        // only upper case english letters
        vector<int> mp(26, 0);
        for(char &ch : tasks) {
            mp[ch-'A']++;
        }   

        // sort the frequency map
        sort(mp.begin(), mp.end());

        int maxFreq = mp[25];
        int space = maxFreq - 1;
        int idleSlots = space * n;

        for(int i=24;i>=0;i--) { // start from second largest
            idleSlots -= min(mp[i], space);
        }
        if(idleSlots > 0)
            return size + idleSlots;

        return size;
    }
};
```
