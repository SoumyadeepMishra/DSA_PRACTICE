# Maximum Bags With Full Capacity of Rocks

```cpp

class Solution {
public:
    int maximumBags(vector<int>& capacity, vector<int>& rocks, int additionalRocks) {
        int n = rocks.size();

        // make an additional array of differenece
        vector<int> diff(n, 0);
        for(int i=0;i<n;i++) {
            diff[i] = capacity[i] - rocks[i];
        }

        // sort korte hobe jate sob theke kom requirement wala age rock pay
        sort(diff.begin(), diff.end());
        int count = 0;

        for(int &num : diff) {
            if(num == 0)
                count++; // full capacity
            else {
                // not zero
                if(additionalRocks >= num) {
                    additionalRocks -= num;
                    count++;
                }
                else // no more additional rocks to fillcapacity
                    break;
            }
        }
        return count;
    }
};
```
