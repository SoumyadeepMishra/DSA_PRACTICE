# Find Polygon With the Largest Perimeter

```cpp

class Solution {
public:
    long long largestPerimeter(vector<int>& nums) {
        int n = nums.size();

        // sort the array
        sort(nums.begin(), nums.end());

        long long currSum = 0;
        long long ans = -1;

        for(int i=0;i<n;i++) {

            if(nums[i] < currSum) {
                ans = currSum + nums[i];
            } 

            currSum += nums[i];
        }

        return ans;
    }
};
```
