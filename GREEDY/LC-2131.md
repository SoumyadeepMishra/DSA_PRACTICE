# Longest Palindrome by Concatenating Two Letter Words

```cpp

class Solution {
public:
    int longestPalindrome(vector<string>& words) {
        int n = words.size();

        // map e frequency store korbo
        unordered_map<string, int> mp;
        for(auto &word : words) {
            mp[word]++;
        }

        bool oneTime = false;
        int result = 0;

        // traverse korte hobe total vector e 
        for(auto &word : words) {
            string rev = word;
            reverse(rev.begin(), rev.end());

            // jodi aasol rr reverseed string same na hoy
            if(word != rev) {
                if(mp[word] > 0 && mp[rev] > 0) {
                    result += 4; // ab     ba
                    mp[word]--;
                    mp[rev]--;
                }
            }
            else { // duto soman --> "aa" "aa"
                if(mp[word] >= 2) {
                    result += 4; // abaa  aaba
                    mp[word] -= 2;
                }
                else if(mp[word] == 1 && oneTime == false) {
                    oneTime = true;
                    result += 2; // abaa cc aaba
                    mp[word]--;
                }
            }
        }
        return result;
    }
};
```
