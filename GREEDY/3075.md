# Maximize Happiness of Selected Children

```cpp

class Solution {
public:
    long long maximumHappinessSum(vector<int>& happiness, int k) {
        long long n = happiness.size();
        
        // sort in descending order
        sort(happiness.begin(), happiness.end(), greater<int>());
        
        long long result = 0;
        long long count  = 0;

        for(int i=0;i<k;i++) {
            
            long long inter = happiness[i] - count;
            if(inter > 0) {
                result += inter;
            }
            count++;
            
        }

        return result;
    }
};

```

# Using max heap (priority queue)

```cpp

class Solution {
public:
    long long maximumHappinessSum(vector<int>& happiness, int k) {
        long long n = happiness.size();

        // declaring max heap
        priority_queue<int> maxHeap;
        for(int i=0;i<n;i++) {
            maxHeap.push(happiness[i]);
        }

        long long result = 0;
        long long count  = 0;
        
        while(k--) {
            int element = maxHeap.top();
            maxHeap.pop();

            long long interMediate = element - count;
            if(interMediate > 0) {
                result += interMediate;
            }
            count++;

        }

        return result;
    }
};
```
