#  Broken Calculator

```cpp

class Solution {
public:
    int brokenCalc(int startValue, int target) {
        // solving from backward
        if(startValue >= target)    
            return startValue-target;

        // if target value is even
        if(target % 2 == 0) {
            return 1 + brokenCalc(startValue, target/2);
        }

        // if target is odd
        return 1 + brokenCalc(startValue, target+1); // 1 + is the operation count
    }
};

```

# Iterative Solution

```cpp

class Solution {
public:
    int brokenCalc(int startValue, int target) {
        // iterative solution 
        // start from the target
        if(startValue >= target) 
            return startValue - target;

        int count = 0;

        while(target > startValue) {
            // if target value is even 
            if(target % 2 == 0){
                target =  target / 2;
                count += 1;
            }
            else {
            target = target + 1;
            count += 1;
            }
            
        }

        // check if target < startValue 
        if(target < startValue)
            count += startValue-target;

        return count;
    }
};
```
