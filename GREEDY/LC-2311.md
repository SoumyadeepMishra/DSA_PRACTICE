# Longest Binary Subsequence Less Than or Equal to K

```cpp

class Solution {
public:
    int longestSubsequence(string s, int k) {
        int n = s.size();

        long long powerValue = 1;
        int length = 0;

        for(int i=n-1;i>=0;i--) {

            if(s[i] == '0') {
                length++;
            }
            // jodi s[i] == 1 hoy
            else if(powerValue <= k) {
                k = k - powerValue;
                length++;
            }

            //updating powerValue
            if(powerValue <= k) {
                powerValue = powerValue * 2;
            }

        }

        return length;
    }
};
```
