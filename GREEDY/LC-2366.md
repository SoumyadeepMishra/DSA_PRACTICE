# Minimum Replacements to Sort the Array

```cpp

class Solution {
public:
    long long minimumReplacement(vector<int>& nums) {
        int n = nums.size();
        long long answer = 0;

        // traverse from back side n-2th element
        for(int i=n-2;i>=0;i--) {
            if(nums[i] <= nums[i+1])
                continue;
            
            long long parts = nums[i] / nums[i+1];
            if(nums[i] % nums[i+1] != 0) {
                parts = parts + 1;
            }
            long long operations = parts - 1;
            answer += operations;

            // there may be more than one disturbance in an array from right side 
            // nums[i] should be divided 
            nums[i] = nums[i] / parts;

        }
        return answer;

    }
};
```
