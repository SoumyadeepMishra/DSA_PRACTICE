class Solution {
public:
    TreeNode* replaceValueInTree(TreeNode* root) {
        if(root == nullptr) return nullptr;

        // bfs for storing the each level sum
        queue<TreeNode*> que;
        que.push(root);
        vector<int> levelSum;

        // step 1 find out each level sum
        while(!que.empty()){
            int n = que.size();
            int currLevelSum = 0;

            while(n--){
                TreeNode* temp = que.front();
                que.pop();

                currLevelSum += temp->val;
                if(temp->left) que.push(temp->left);
                if(temp->right) que.push(temp->right);
                
            }
            levelSum.push_back(currLevelSum);
        }

        // step 2 (update each node with cousin sum)
        que.push(root);
        root->val = 0;
        int reqLvl = 1;

        while(!que.empty()){
            int n = que.size();
            while(n--){
                TreeNode* temp = que.front();
                que.pop();

            // level sum of current level - siblingSum
                int siblingSum = 0;
                if(temp->left != nullptr){
                    siblingSum += temp->left->val;
                }
                if(temp->right != nullptr){
                    siblingSum += temp->right->val;
                }

                if(temp->left){
                    temp->left->val = levelSum[reqLvl] - siblingSum;
                    que.push(temp->left);
                }
                if(temp->right){
                    temp->right->val = levelSum[reqLvl] - siblingSum;
                    que.push(temp->right);
                }
            }
            reqLvl++;
        }

        return root;
    }
};
