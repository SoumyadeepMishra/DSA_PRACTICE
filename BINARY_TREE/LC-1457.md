# Pseudo pallindromic path in a Binary Tree

code 

```cpp

class Solution {
public:
    int result = 0;

    void solve(TreeNode* root, vector<int>& count){
        if(root == nullptr) return;

        count[root->val] ++;

        // reach the leaf node
        if(root->left == nullptr && root->right == nullptr){
            int oddFreq = 0;
            // one path is completed now iterate through the vector
            for(int i=0;i<=9;i++){
                if(count[i]%2 != 0){
                    oddFreq ++;
                }
            }

            if(oddFreq <= 1){
                result += 1;
            }
        }

        solve(root->left, count);
        solve(root->right, count);

        count[root->val] --;
    }

    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> count(10, 0);
        solve(root, count);
        return result;
    }
};
```
