# Minimum Number of Operations to Sort a Binary Tree by Level

code

```cpp

class Solution {
public:

    int countMinSwapsToSort(vector<int> vec){
        unordered_map<int, int> mp; // nums[i] -> i
        int n = vec.size();
        // populating the map
        for(int i=0;i<n;i++){
            mp[vec[i]] = i;
        }

        vector<int> sortedVec(vec.begin(), vec.end());
        sort(sortedVec.begin(), sortedVec.end()); // vec vector has been sorted

        int swaps = 0;
        
        // actual calculations
        for(int i=0;i<n;i++){
            if(vec[i] == sortedVec[i]){
                continue;
            }

            int currIdx = mp[sortedVec[i]];
            mp[vec[i]] = currIdx;
            swap(vec[currIdx], vec[i]);
            swaps ++; 
        }
        return swaps;
    }

    int minimumOperations(TreeNode* root) {
        queue<TreeNode*> que;
        que.push(root);

        int result = 0;

        while(!que.empty()){
            int n = que.size();
            vector<int> vec; // all values of the level

            while(n--){
                TreeNode* temp = que.front();
                que.pop();

                vec.push_back(temp->val);

                if(temp->left) que.push(temp->left);
                if(temp->right) que.push(temp->right);
            }

            result += countMinSwapsToSort(vec);
            
        }

        return result;
    }
};
```
