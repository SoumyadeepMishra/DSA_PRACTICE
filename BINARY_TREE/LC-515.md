# Find largest value in Each Tree row

code(BFS)

```CPP

class Solution {
public:
    vector<int> largestValues(TreeNode* root) {
        if(root == nullptr) return {};
        vector<int> result;
        queue<TreeNode*> que;
        que.push(root);

        while(!que.empty()){
            int n = que.size();
            int maxEl = INT_MIN;

            while(n--){
                TreeNode* temp = que.front();
                que.pop();
                
                maxEl = max(maxEl, temp->val);

                if(temp->left)
                    que.push(temp->left);
                if(temp->right)
                    que.push(temp->right);
            }
            result.push_back(maxEl);
        }
        return result;
    }
};
```
