# Path Sum

class Solution {
public:

    bool preOrder(TreeNode* root, int sum, int targetSum){
        if(root == nullptr) return false;

        sum += root->val;
        // check if we have reached the last node
        if(root->left == nullptr && root->right == nullptr){
            if(sum == targetSum) return true;
            return false;
        }
        bool left = preOrder(root->left, sum, targetSum);
        bool right = preOrder(root->right, sum, targetSum);

        return left || right;
    }

    bool hasPathSum(TreeNode* root, int targetSum) {
        int sum = 0;
        bool result = preOrder(root, sum, targetSum);
        return result;
    }
};
