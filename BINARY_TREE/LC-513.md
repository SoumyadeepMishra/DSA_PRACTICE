# Find bottom left Tree value

code(DFS)

```cpp

class Solution {
public:
    int maxDepth;
    int result;

    void dfs(TreeNode* root, int currDepth){
        if(root == nullptr) return;

        // main condition
        if(currDepth > maxDepth){
            maxDepth = currDepth;
            result = root->val;
        }

        dfs(root->left, currDepth+1);
        dfs(root->right, currDepth+1);
    }

    int findBottomLeftValue(TreeNode* root) {
        maxDepth = -1;
        dfs(root, 0);
        return result;
    }
};

```
