# Smallest string starting from leaf 

code(DFS)

```cpp

class Solution {
public:
    string result = "";

    void dfs(TreeNode* root, string curr){
        if(root == nullptr) return;

        curr = char(root->val + 'a') + curr;

        if(root->left == nullptr && root->right == nullptr){
            if(result == "" || result > curr){
                result = curr;
            }
            return;
        }

        dfs(root->left, curr);
        dfs(root->right, curr);
    }

    string smallestFromLeaf(TreeNode* root) {
        string curr = "";
        dfs(root, curr);
        return result;
    }
};

```

code(BFS)

```cpp

class Solution {
public:
    string smallestFromLeaf(TreeNode* root) {
        // bfs
        queue<pair<TreeNode*, string>> que;
        que.push({root, string(1, char(root->val +'a'))});

        string result = "";

        while(!que.empty()){
            auto[temp, curr] = que.front();
            que.pop();

            if(temp->left == nullptr && temp->right == nullptr){
                if(result == "" || result > curr){
                    result = curr;
                }
            }

            if(temp->left){
                que.push({temp->left, char(temp->left->val + 'a') + curr});
            }
            if(temp->right){
                que.push({temp->right, char(temp->right->val + 'a') + curr});
            }
        }

        return result;
    }
};
```
