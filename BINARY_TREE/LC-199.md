class Solution {
public:
    vector<int> rightSideView(TreeNode* root) {
        if(root == nullptr) return {};
        // level order traversal
        queue<TreeNode*> q;
        q.push(root);

        vector<int>result;

        while(!q.empty()){
            int n = q.size();
            TreeNode* temp;
            while(n--){
                temp = q.front();
                q.pop();
                if(temp->left) q.push(temp->left);
                if(temp->right) q.push(temp->right);
            }
            // temp me uss level ka last element rahega
            result.push_back(temp->val);
        }
        return result;
    }
};
