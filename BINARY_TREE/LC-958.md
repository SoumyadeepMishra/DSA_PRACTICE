Check Completeness of a Binary Tree

code 

```cpp

class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        // using BFS
        if(!root) return true;
        queue<TreeNode*> que;
        que.push(root);

        bool pastNULL = false; // past me null node dekha hai ki nahi

        while(!que.empty()){
            TreeNode* node = que.front();
            que.pop();

            if(node == nullptr){
                pastNULL = true;
            }
            else{
                
                if(pastNULL == true) return false;

                que.push(node->left);
                que.push(node->right);

            }
            
        }
        return true;
    }
};
```
