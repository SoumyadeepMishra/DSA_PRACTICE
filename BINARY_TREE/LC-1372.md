# Longest ZigZag Path in a Binary Tree

code

```cpp

class Solution {
public:
    int maxPath = 0;

    void solve (TreeNode* root, int steps, bool goLeft){
        if(root == nullptr) return;

        maxPath = max(maxPath, steps);

        if(goLeft == true){
            solve(root->left, steps+1, false); // then goto right
            solve(root->right, 1, true); // then goto left
        } else{
            solve(root->right, steps+1, true);
            solve(root->left, 1, false);
        }
    }

    int longestZigZag(TreeNode* root) {
        solve(root, 0, true); // goto left
        solve(root, 0, false); // goto right

        return maxPath;
    }
};

```

Better Approach

```cpp

class Solution {
public:
    int maxPath = 0;

    void helper(TreeNode* root, int left, int right){
        if(root == nullptr) return;

        maxPath = max({maxPath, left, right});

        helper(root->left, right+1, 0);
        helper(root->right, 0, left+1);
    }

    int longestZigZag(TreeNode* root) {
        helper(root, 0, 0);

        return maxPath;
    }
};
```
