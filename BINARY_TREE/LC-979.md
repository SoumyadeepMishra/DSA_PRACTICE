# Distribute coins in binary Tree

code

```cpp

class Solution {
public:

    int countMoves(TreeNode* root, int &moves){
        if(root == nullptr) return 0;

        int l = countMoves(root->left, moves);
        int r = countMoves(root->right, moves);

        moves += abs(l) + abs(r); // here is the main funda

        return (l + r + root->val) - 1; // here also main funda
    }

    int distributeCoins(TreeNode* root) {
        int moves = 0;
        countMoves(root, moves);
        return moves;
    }
};
```
