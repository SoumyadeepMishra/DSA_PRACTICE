# Maximum Width of Binary Tree

code

```cpp

class Solution {
public:
    typedef unsigned long long ll; // outofbound problem solved 
    int widthOfBinaryTree(TreeNode* root) {
        queue<pair<TreeNode*, ll>> que;
        que.push({root, 0});

        ll maxWidth = 0;
        while(!que.empty()){
            ll L = que.front().second;
            ll R = que.back().second;

            maxWidth = max(maxWidth, R-L+1);
            int n = que.size();

            while(n--){
                TreeNode* temp = que.front().first;
                ll idx = que.front().second;
                que.pop();

                if(temp->left != nullptr) que.push({temp->left, 2*idx+1});
                if(temp->right != nullptr) que.push({temp->right, 2*idx+2});
            }
        }

        return maxWidth;
    }
};
```
