## Next Permutation

```cpp

class Solution {
public:

    void rev(vector<int> &nums, int start, int end){
        while(start <= end){
            swap(nums[start], nums[end]);
            start++;
            end--;
        }
        return;
    }

    void nextPermutation(vector<int>& nums) {
        int n = nums.size();
        // search for the pivot index --> 1,2,5,4,3 here pivot = 2

        int pivot = -1;
        for(int i=n-2;i>=0;i--){
            if(nums[i] < nums[i+1]){
                pivot = i;
                break;
            }
        }
        // if no pivot means 5,4,3,2,1 like array
        if(pivot == -1){
             return rev(nums, 0, n-1);
        }

        // step 2
        for(int i=n-1;i>pivot;i--){
            if(nums[i] > nums[pivot]){
                swap(nums[i],nums[pivot]);
                break;
            }
        }

        // step 3 reverse from pivot+1 to n-1
        return rev(nums, pivot+1, n-1);

    }
};
```
