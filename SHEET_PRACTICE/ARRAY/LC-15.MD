## 3 sum 

```cpp

class Solution {
public:
    vector<vector<int>> result;

    void two_sum(vector<int> &nums, int target, int i, int j) {
        // applying two sum 
        while(i < j) {
            if(nums[i] + nums[j] < target) i++;
            else if( nums[i] + nums[j] > target) j--;

            // equal to target
            else {
                while(i < j && nums[i]==nums[i+1]) i++;
                while(i < j && nums[j]==nums[j-1]) j--;

                result.push_back({-target,nums[i],nums[j]});
                i++;
                j--;
            }
        }
        return;
    }

    vector<vector<int>> threeSum(vector<int>& nums) {
        result.clear();

        int n = nums.size();

        if(n<3) return {};
        // first step is to sort the array to apply teo pointer
        sort(nums.begin(), nums.end());

        // second step is to fix one target element through loop
        for(int i=0;i<n;i++){
            if(i>0 && nums[i] == nums[i-1]) { // when i=0 then i-1 will give error
                continue;
            }
            int target = -nums[i];
            // call two sum for the remaining array
            two_sum(nums, target, i+1, n-1);
        }
        return result;
    }
};
```
