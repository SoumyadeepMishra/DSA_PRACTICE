## search in a 2D matrix


```cpp

class Solution {
public:

    bool search_in_row(vector<vector<int>> &mat, int target, int targetRow){
        int n = mat[0].size(); // for column size

        int start = 0, end = n-1;
        while(start <= end){
            int mid = start + (end - start)/2;

            if(target == mat[targetRow][mid]) return true;
            else if(target > mat[targetRow][mid]) start = mid + 1;
            else end = mid - 1;

        }

        return false;
    }

    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size(), n = matrix[0].size();

        // have to search in the row range 
        int startRow = 0, endRow = m-1;
        // binary search on row 
        while(startRow <= endRow){
            int mid = startRow + (endRow - startRow)/2;
            
            if(matrix[mid][0] <= target && target <= matrix[mid][n-1]){
                // found the target row to search for target --> binary search
                return search_in_row(matrix, target, mid);

            } else if(target >= matrix[mid][n-1]){
                // target is in the down row 
                startRow = mid + 1;
            } else{
                // target is in the up row 
                endRow = mid - 1;
            }
        }
        return false;
    }
};

```
