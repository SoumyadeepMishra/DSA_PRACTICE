# Set Matrix Zero

```cpp

class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        int m = matrix.size(); // row size
        int n = matrix[0].size(); // col size

        vector<bool> row(m, false);
        vector<bool> col(n, false);

        for(int i=0;i<m;i++) {
            for(int j=0;j<n;j++) {
                if(matrix[i][j] == 0) {
                    row[i] = true;
                    col[j] = true;
                }
            }
        }

        for(int i=0;i<m;i++) {
            for(int j=0;j<n;j++) {
                if(row[i] == true || col[j] == true) {
                    matrix[i][j] = 0;
                }
            }
        }
    }
};
```

```cpp

class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();

        // start from 1th row and 1th column solve 0th row and column separately for corner cases
        // handle first row and column separately
        int firstRowImpacted = false;
        int firstColImpacted = false;

        for(int i=0;i<n;i++) {
            if(matrix[0][i] == 0) {
                firstRowImpacted = true;
                break;
            }
        }

        for(int i=0;i<m;i++) {
            if(matrix[i][0] == 0) {
                firstColImpacted = true;
                break;
            }
        }

    // indicator value setting
        for(int i=1;i<m;i++) {
            for(int j=1;j<n;j++) {
                if(matrix[i][j] == 0) {
                    matrix[i][0] = 0;
                    matrix[0][j] = 0;
                }
            }
        }

        for(int i=1;i<m;i++) {
            for(int j=1;j<n;j++) {
                if(matrix[i][0] == 0 || matrix[0][j] == 0) {
                    matrix[i][j] = 0;
                }
            }
        }

        

        // set accordingly to the first row and column
        if(firstRowImpacted) {
            for(int i=0;i<n;i++) {
                matrix[0][i] = 0;
            }
        }

        if(firstColImpacted) {
            for(int i=0;i<m;i++) {
                matrix[i][0] = 0;
            }
        }
    }
};
```
