# Boats to save people

code

```cpp

class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int n = people.size();
        sort(people.begin(), people.end());

        int st = 0, end = n-1;
        int boatCnt = 0;

        while(st <= end){
            if(people[st]+people[end] <= limit){
                st++;
                end--;
                boatCnt++;
            }
            else if(people[end] > limit){
                end--;
            }
            else{
                end--;
                boatCnt++;
            }
        }

        return boatCnt;
    }
};
```
