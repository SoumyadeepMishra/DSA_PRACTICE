# Minimize maximum pair sum in Array

code

```cpp

class Solution {
public:
    int minPairSum(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(), nums.end()); // main step dhyan rakhte hobe

        int result = 0;

        int l = 0, r = n-1;

        while(l<r){
            int sum = nums[l] + nums[r];
            result = max(result, sum);
            l++;
            r--;
        }
        return result;
    }
};
```
