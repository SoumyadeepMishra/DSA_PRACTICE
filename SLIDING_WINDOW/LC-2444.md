## Count Subarrays With Fixed Bounds

```cpp

class Solution {
public:
    long long countSubarrays(vector<int>& nums, int minK, int maxK) {
        long long ans = 0;

        int last_min = -1;
        int last_max = -1;
        int culprit_idx = -1;

        for(int i=0;i<nums.size();i++){
            // main concept of this question
            if(nums[i]<minK || nums[i]>maxK)
                culprit_idx = i;
            if(nums[i]==minK)
                last_min = i;
            if(nums[i]==maxK)
                last_max = i;
            
            long long smaller = min(last_min, last_max);
            long long temp = smaller - culprit_idx;

            if(temp<=0){
                ans += 0;
            } else{
                ans += temp;
            }
        }
        return ans;
    }
};
```
