## Longest Subarray of 1's After Deleting One Element

```cpp

class Solution {
public:
    int longestSubarray(vector<int>& nums) {
        int n = nums.size();
        int countZero = 0;
        int result = 0;

        int i = 0;
        int j = 0;
        while(j < n){
            if(nums[j] == 0){
                countZero++;
            }
            
            while(countZero > 1){
                if(nums[i] == 0) countZero--;
                i++;
            }
            result = max(result, j-i);
             j++;
        }
        return result;
    }
};
```
