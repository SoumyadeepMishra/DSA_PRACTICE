# Find the Power of K-Size Subarrays I

```cpp

class Solution {
public:
    vector<int> resultsArray(vector<int>& nums, int k) {
        int n = nums.size();

        int count = 1;
        for(int i=1;i<k;i++) {
            if(nums[i-1] + 1 == nums[i])
                count++;
            else 
                count = 1;
        }

        vector<int> result(n-k+1, -1);
        if(count == k)
            result[0] = nums[k-1];

        int i = 1;
        int j = k;
        while(j < n) {

            if(nums[j-1] + 1 == nums[j]) {
                count++;
            }
            else {
                count = 1;
            }

            if(count >= k) {
                result[i] = nums[j]; // j is the current max element
            }

            i++;
            j++;

        }

        return result;
    }
};
```
