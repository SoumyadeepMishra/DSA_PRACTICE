# Count Complete Subarrays in an Array

```cpp

class Solution {
public:
    int countCompleteSubarrays(vector<int>& nums) {
        int n = nums.size();

        // set to track unique elements
        unordered_set<int> st(nums.begin(), nums.end());
        int unique = st.size();

        unordered_map<int,int> mp;
        int i = 0;
        int j = 0;
        int result = 0;
        while(j < n) {

            mp[nums[j]]++;

            while(mp.size() == unique) {
                result += (n - j);
                mp[nums[i]]--;
                if(mp[nums[i]] == 0) {
                    mp.erase(nums[i]);
                }
                i++;
            }

            j++;

        }

        return result;
    }
};
```
