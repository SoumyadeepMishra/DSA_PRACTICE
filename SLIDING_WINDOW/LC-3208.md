# Alternating Groups II

```cpp

class Solution {
public:
    int numberOfAlternatingGroups(vector<int>& colors, int k) {
        int n = colors.size();
        int N = n + (k - 1);
        for(int i=0;i<=k-1;i++) {
            colors.push_back(colors[i]);
        }

        int i = 0;
        int j = 1;
        int result = 0;

        while(j < N) {

            if(colors[j-1] == colors[j]) {
                i = j;
                j++;
                continue;
            }

            if(j-i+1 == k) {
                result++;
                i++;
            }

            j++;

        }

        return result;
    }
};
```

# single for loop solution 

```cpp

class Solution {
public:
    int numberOfAlternatingGroups(vector<int>& colors, int k) {
        int n = colors.size();

        int length = 1;
        int lastColor = colors[0];
        int result = 0;

        for(int i=1;i<n;i++) {

            if(colors[i] == lastColor) {
                length = 1;
                lastColor = colors[i];
                continue;
            }

            length++;

            if(length >= k) {
                result++;
            }

            lastColor = colors[i];
        }

        for(int i=0;i<k-1;i++) {
            if(colors[i] == lastColor) {
                length = 1;
                lastColor = colors[i];
                continue;
            }

            length++;

            if(length >= k) {
                result++;
            }

            lastColor = colors[i];
        }

        return result;
    }
};
```
