#  Number of Smooth Descent Periods of a Stock

```cpp

class Solution {
public:
    long long getDescentPeriods(vector<int>& prices) {
        int n = prices.size();

        long long count = 1;
        long long ans = 1;
        int prev = prices[0];

        for(int i=1;i<n;i++) {

            if(prev - prices[i] == 1) {
                count++;
            }
            else {
                count = 1;
            }
            ans += count;
            prev = prices[i];
        }

        return ans;
    }
};
```
