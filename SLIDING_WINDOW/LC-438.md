## Find All Anagrams in a String

```cpp

class Solution {
public:

    bool allzero(vector<int> &counter){
        for(int &num : counter){
            if(num != 0) return false;
        }
        return true;
    }

    vector<int> findAnagrams(string s, string p) {
        int n = s.length();
        // to store the frequency of charcters of p string 
        vector<int> counter(26, 0);
        for(int i=0;i<p.length();i++){
            char ch = p[i];
            int idx = p[i] - 'a';
            counter[idx]++;
        }
        // apply sliding window technique
        vector<int> result;
        int i=0, j=0;
        while(j < n){

            counter[s[j]-'a']--;

            if(j-i+1 == p.length()){
                if(allzero(counter)){
                    result.push_back(i);
                }
                counter[s[i]-'a']++;
                i++;
            }
            j++;
        }
        return result;
    }
};

```
