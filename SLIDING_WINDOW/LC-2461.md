# Maximum Sum of Distinct Subarrays With Length K

```cpp

class Solution {
public:
    long long maximumSubarraySum(vector<int>& nums, int k) {
        int n = nums.size();

        long long result = 0;
        long long sum = 0;

        int i = 0;
        int j = 0;

        // set to track the duplicate element
        unordered_set<int> st;
        while(j < n) {

            // check if already present in the set or not
            while(st.count(nums[j])) {
                sum -= nums[i];
                st.erase(nums[i]);
                i++;
            }

            sum += nums[j];
            st.insert(nums[j]);

            if(j - i + 1 == k) {
                result = max(result, sum);
                sum -= nums[i];
                st.erase(nums[i]);
                i++;
            }

            j++;
        }

        return result;
    }
};
```
