# Permutation in String

```cpp

class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int n = s1.size();
        int m = s2.size();

        if(n > m)
            return false;

        // substring approach
        sort(s1.begin(), s1.end());

        for(int i=0;i<=m-n;i++) {

            string substr = s2.substr(i, n); // n length er substring nilam
            // sort korbo
            sort(substr.begin(), substr.end());

            if(s1 == substr)
                return true;

        }

        return false;
    }
};
```

# sliding window

```cpp

class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int n = s1.size();
        int m = s2.size();

        vector<int> s1Freq(26, 0);
        for(auto &ch : s1) {
            s1Freq[ch-'a']++;
        }

        vector<int> s2Freq(26, 0);

        int i = 0;
        int j = 0;
        while(j < m) {

            s2Freq[s2[j]-'a']++;

            if(j-i+1 > n) {
                s2Freq[s2[i]-'a']--;
                i++;
            }

            // main condition
            if(s1Freq == s2Freq)
                return true;

            j++;

        }

        return false;
    }
};
```
