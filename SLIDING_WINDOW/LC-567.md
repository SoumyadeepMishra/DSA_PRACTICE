# Permutation in String

```cpp

class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int n = s1.size();
        int m = s2.size();

        if(n > m)
            return false;

        // substring approach
        sort(s1.begin(), s1.end());

        for(int i=0;i<=m-n;i++) {

            string substr = s2.substr(i, n); // n length er substring nilam
            // sort korbo
            sort(substr.begin(), substr.end());

            if(s1 == substr)
                return true;

        }

        return false;
    }
};
```
