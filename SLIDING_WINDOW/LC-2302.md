# Count Subarrays With Score Less Than K

```cpp

class Solution {
public:
    long long countSubarrays(vector<int>& nums, long long k) {
        long long cnt=0;
        long long sum=0;
        long long l=0;

        long long n=nums.size();
        for(int i=0;i<n;i++){
            sum+=nums[i];
            while(l<=i && sum* (i-l+1)>=k){
                sum-=nums[l];
                l++;
            }
            cnt+=(i-l+1);
        }
       return  cnt;
        
    }
};
```
