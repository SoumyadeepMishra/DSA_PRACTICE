# Minimum Swaps to Group All 1's Together II

```cpp

class Solution {
public:
    int minSwaps(vector<int>& nums) {
        int n = nums.size();

        int i = 0;
        int j = 0;
        int currOnes = 0;
        int swaps  = INT_MAX;

        // count total ones in array. As window size would be total number of ones
        int totalOnes = 0;
        for(auto &num : nums) {
            if(num == 1)
                totalOnes++;
        }

        while(j < 2*n) {

            if(nums[j%n] == 1)
                currOnes++;

            if(j-i+1 > totalOnes) {
                if(nums[i%n] == 1)
                    currOnes -= nums[i%n];
                i++;
            }

            swaps = min(swaps, totalOnes-currOnes);
            j++;
        }

        return swaps;
    }
};
```
