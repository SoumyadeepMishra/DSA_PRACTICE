# Count Substrings That Satisfy K-Constraint I

```cpp

class Solution {
public:
    int countKConstraintSubstrings(string s, int k) {
        int n = s.size();

        int i = 0;
        int j = 0;
        int countZero = 0;
        int countOne = 0;
        int result = 0;

        while(j < n) {

            if(s[j] == '0')
                countZero++;
            else 
                countOne++;

            while(countZero > k && countOne > k) {
                if(s[i] == '0')
                    countZero--;
                else 
                    countOne--;

                i++;
            }

            result += (j-i+1);
            j++;

        }

        return result;
    }
};
```
