# Grumpy Bookstore Owner

class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int minutes) {
        int n = customers.size();

        int i = 0;
        int j = minutes - 1;
        int currUnsatisfied = 0;

        for(int i=0;i<=minutes-1;i++) {
            if(grumpy[i] == 1)
                currUnsatisfied += customers[i];
        }
        
        int maxUnsatisfied = currUnsatisfied; // window created 

        j++; // expanding the window 

        while(j < n) {

            if(grumpy[j] == 1) {
                currUnsatisfied += customers[j];
            }
            if(grumpy[i] == 1) {
                currUnsatisfied -= customers[i];
            }

            maxUnsatisfied = max(maxUnsatisfied, currUnsatisfied);
            i++;
            j++;

        }

        int total = maxUnsatisfied; // max in minutes sized window are satisfied using power
        for(int i=0;i<n;i++) {
            if(grumpy[i] == 0)
                total += customers[i];
        }

        return total;
    }
};
