## Maximize the Confusion of an Exam

```cpp

class Solution {
public:
    int maxConsecutiveAnswers(string answerKey, int k) {
        // we have two cases here
        
        // case-1 ("F"->"T")
        int n = answerKey.length();
        int i=0, j=0;
        int countF = 0;
        int result_1 = 0;

        while(j < n){
            if(answerKey[j] == 'F'){
                countF++;
            }
            while(countF > k){
                if(answerKey[i] == 'F') countF--;
                i++;
            }
            result_1 = max(result_1 , j-i+1);
            j++;
        }

        // case-2 ("T"->"F")
         i=0, j=0;
        int countT = 0;
        int result_2 = 0;

        while(j < n){
            if(answerKey[j] == 'T'){
                countT++;
            }
            while(countT > k){
                if(answerKey[i] == 'T') countT--;
                i++;
            }
            result_2 = max(result_2 , j-i+1);
            j++;
        }

        return max(result_1,result_2);
    }
};
```
