# Count Subarrays Where Max Element Appears at Least K Times

```cpp

class Solution {
public:
    long long countSubarrays(vector<int>& nums, int k) {
        int n = nums.size();
        // max element of the array
        int maxE = *max_element(nums.begin(), nums.end());

        int i = 0;
        int j = 0;
        long long countMax = 0;
        long long result = 0;

        while(j < n) {

            if(nums[j] == maxE)
                countMax++;

            while(countMax >= k) {
                result += n - j;

                if(nums[i] == maxE)
                    countMax--;

                i++;
            }

            j++;
        }

        return result;
    }
};
```
