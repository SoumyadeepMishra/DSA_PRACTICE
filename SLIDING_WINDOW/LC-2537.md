# Count the Number of Good Subarrays

```cpp

class Solution {
public:
    long long countGood(vector<int>& nums, int k) {
        int n = nums.size();

        int i = 0;
        int j = 0;
        long long countPair = 0;
        long long ans = 0;

        unordered_map<int, int> mp;

        while(j < n) {

            countPair += mp[nums[j]];
            mp[nums[j]]++;

            while(countPair >= k) {
                ans += (n - j);
                mp[nums[i]]--;
                countPair -= mp[nums[i]];
                i++;
            }

            j++;

        }

        return ans;
    }
};
```
