# Alternating Groups I

```cpp

class Solution {
public:
    int numberOfAlternatingGroups(vector<int>& colors) {
        int n = colors.size();
        int N = n + 2;
        for(int i=0;i<2;i++) {
            colors.push_back(colors[i]);
        }
        
        int result = 0;

        for(int i=1;i<N-1;i++) {

            if(colors[i-1] == colors[i] || colors[i] == colors[i+1]) {
                continue;
            }

            result++;

        }

        return result;
    }
};
```
