# Partition Equal Subset Sum

```cpp

class Solution {
public:

    bool subsetSum(vector<int> &nums, int sum, int n) {

        vector<vector<int>> dp(n+1, vector<int> (sum+1));

        // initialization
        for(int i=0;i<=n;i++) {
            for(int j=0;j<=sum;j++) {
                
                if(i == 0 && j == 0)
                    dp[i][j] = true;
                else if(i == 0)
                    dp[i][j] = false;
                else if(j == 0)
                    dp[i][j] = true;

            }
        }

        // main condition
        for(int i=1;i<=n;i++) {
            for(int j=1;j<=sum;j++) {

                if(nums[i-1] <= j) {
                    dp[i][j] = dp[i-1][j-nums[i-1]] || dp[i-1][j];
                }
                else {
                    dp[i][j] = dp[i-1][j];
                }

            }
        }

        return dp[n][sum];

    }

    bool canPartition(vector<int>& nums) {
        // first check total sum of the array element 
        int n = nums.size();
        int sum = 0;
        for(int i=0;i<n;i++) {
            sum += nums[i];
        }

        if(sum%2 != 0)
            return false;

        else if(sum%2 == 0)
            return subsetSum(nums, sum/2, n);

        return 0;
    }
};
```
