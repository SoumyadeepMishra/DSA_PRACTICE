# Number of Substrings With Only 1s

```cpp

class Solution {
public:
    int mod = 1e9 + 7;
    int numSub(string s) {
        int n = s.size();

        int length = 0;
        int result = 0;

        for(int i=0;i<n;i++) {
            
            if(s[i] == '1') {
                length++;
            }
            else {
                length = 0;
            }

            result = (result + length) % mod;

        }

        return result;
    }
};

```
