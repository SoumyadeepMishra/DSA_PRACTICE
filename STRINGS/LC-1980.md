# Find Unique Binary String

```cpp

class Solution {
public:
    int n;
    unordered_set<string> numset;

    string generate(string curr) {

        if(curr.length() == n) {
            if(!numset.count(curr)) {
                return curr;
            }
            return "";
        }

        // try with '0'
        string temp = generate(curr + "0");
        if(temp != "") {
            return temp;
        }
        // try with "1"
        temp = generate(curr + "1");
        if(temp != "") {
            return temp;
        }

        return temp;

    }

    string findDifferentBinaryString(vector<string>& nums) {
        n = nums.size();
        for(string s : nums) {
            numset.insert(s);
        }

        string s = "";
        return generate(s);
    }
};

```

# check 0th char of 1st string then 1st char of 2nd string and so on.... automatically the result will be unique from n strings. Hence it will be the answer

```cpp

class Solution {
public:
    string findDifferentBinaryString(vector<string>& nums) {
        int n = nums.size();
        string result = "";

        for(int i=0;i<n;i++) {

            char ch = nums[i][i];
            if(ch == '0') {
                result += "1";
            }
            else {
                result += "0";
            }

        }

        return result;
    }
};
```
