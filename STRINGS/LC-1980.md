# Find Unique Binary String

```cpp

class Solution {
public:
    int n;
    unordered_set<string> numset;

    string generate(string curr) {

        if(curr.length() == n) {
            if(!numset.count(curr)) {
                return curr;
            }
            return "";
        }

        // try with '0'
        string temp = generate(curr + "0");
        if(temp != "") {
            return temp;
        }
        // try with "1"
        temp = generate(curr + "1");
        if(temp != "") {
            return temp;
        }

        return temp;

    }

    string findDifferentBinaryString(vector<string>& nums) {
        n = nums.size();
        for(string s : nums) {
            numset.insert(s);
        }

        string s = "";
        return generate(s);
    }
};
```
