# Length of the Longest Alphabetical Continuous Substring

```cpp

class Solution {
public:
    int longestContinuousSubstring(string s) {
        int n = s.size();
        int result = 0;
        int length = 0;

        for(int i=0;i<n;i++) {

            if(i > 0 && s[i]-s[i-1] == 1) {
                length++;
            }
            else {
                length = 1;
            }

            result = max(result, length);

        }

        return result;
    }
};
```
