# Consecutive Characters

```cpp

class Solution {
public:
    int maxPower(string s) {
        int n = s.size();

        if(n == 1)
            return 1;
            
        int length = 1;
        int result = 0;

        for(int i=1;i<n;i++) {

           while(s[i] == s[i-1]) {
                length++;
                i++;
           }
            result = max(result, length);
            length = 1;
        }
        return result;
    }
};
```
