# Count the Number of Houses at a Certain Distance I

```cpp

class Solution {
public:
    vector<int> countOfPairs(int n, int x, int y) {
        // floyd warshall algorithm
        // i and i+1 are connected
        vector<vector<int>> matrix(n+1, vector<int>(n+1, 1e9));

        for(int i=1;i<n;i++) {
            matrix[i][i+1] = 1;
            matrix[i+1][i] = 1;
        }

        matrix[x][y] = 1;
        matrix[y][x] = 1;

        // floyd warshall algorithm 
        // start from k = 1
        for(int via=1;via<=n;via++) {

            for(int i=1;i<=n;i++) {
                for(int j=1;j<=n;j++) {
                    
                    if(i == j)
                        matrix[i][j] = 0;

                    else {
                        matrix[i][j] = min(matrix[i][j], matrix[i][via] + matrix[via][j]);
                    }
                    

                }
            }

        }

        vector<int> result(n);
        for(int i=1;i<=n;i++) {
            for(int j=1;j<=n;j++) {

                if(i == j)
                    continue;

                int val = matrix[i][j];
                result[val - 1]++;

            }
        }

        return result;
    }
};
```
