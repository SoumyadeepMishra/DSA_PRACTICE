# Clone Graph

```cpp

/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> neighbors;
    Node() {
        val = 0;
        neighbors = vector<Node*>();
    }
    Node(int _val) {
        val = _val;
        neighbors = vector<Node*>();
    }
    Node(int _val, vector<Node*> _neighbors) {
        val = _val;
        neighbors = _neighbors;
    }
};
*/

void dfs(Node* node, Node* cloneNode, unordered_map<Node*, Node*> &mp) {
    for(Node* n : node->neighbors) {
        // check already available in the map or not 
        // if not 
        if(mp.find(n) == mp.end()) {
            Node* clone = new Node(n->val);
            mp[n] = clone;
            cloneNode->neighbors.push_back(clone);
            dfs(n, clone, mp);
        }
        else {
            // if already in map ... retrieve it from map
            cloneNode->neighbors.push_back(mp[n]);
        }
    }
}

class Solution {
public:
    Node* cloneGraph(Node* node) {
        if(node == nullptr) {
            return nullptr;
        }

        // clone the given node first
        Node* cloneNode = new Node(node->val);

        // store {realNode, cloneNode} in map
        unordered_map<Node*, Node*> mp;
        mp[node] = cloneNode;

        dfs(node, cloneNode, mp);

        return cloneNode;
    }
};
```
