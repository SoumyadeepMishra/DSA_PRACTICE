#  Path with Maximum Probability

```cpp

class Solution {
public:
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start_node, int end_node) {
        // make adjacency list
        unordered_map<int, vector<pair<int, double>>> adj;
        for(int i=0;i<edges.size();i++) {

            int u = edges[i][0];
            int v = edges[i][1];
            double prob = succProb[i];

            adj[u].push_back({v, prob});
            adj[v].push_back({u, prob});

        }

        vector<double> result(n, 0);

        // start dijkstra logic but use max heeap
        priority_queue<pair<double, int>> que; // {prob, node}
        result[start_node] = 1.0;
        que.push({1.0, start_node});

        while(!que.empty()) {
            int node = que.top().second;
            double prob = que.top().first;
            que.pop();

            for(auto &ngh : adj[node]) {
                int nghNode = ngh.first;
                double nghProb = ngh.second;

                // main condition
                if(result[nghNode] < prob*nghProb) {
                    result[nghNode] = prob*nghProb;
                    que.push({prob*nghProb, nghNode});
                }
            }
        }

        return result[end_node];
    }
};
```
