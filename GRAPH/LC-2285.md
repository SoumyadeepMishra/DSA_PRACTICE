# Maximum Total Importance of Roads

```cpp

class Solution {
public:
    long long maximumImportance(int n, vector<vector<int>>& roads) {
        // count degree of each vertex
        // then multiply assigned value with degree
        vector<int> degree(n, 0);
        for(auto &road : roads) {
            
            int u = road[0];
            int v = road[1];

            degree[u]++;
            degree[v]++;

        }

        sort(begin(degree), end(degree));
        long long assignedValue = 1;
        long long sum = 0;

        for(int i=0;i<n;i++) {

            sum += (assignedValue * degree[i]);
            assignedValue++;

        }

        return sum;
    }
};
```
