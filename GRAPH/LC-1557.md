# Minimum Number of Vertices to Reach All Nodes

```cpp

class Solution {
public:
    vector<int> findSmallestSetOfVertices(int n, vector<vector<int>>& edges) {
        // just track the indegree of every vertex
        vector<bool> indegree(n, false);
        for(auto &edge : edges) {
            int u = edge[0];
            int v = edge[1];

            // u --> v
            indegree[v] = true;
        }

        // traverse to check which node has no indegree
        vector<int> result;

        for(int i=0;i<n;i++) {
            if(indegree[i] == false) {
                result.push_back(i);
            }
        }

        return result;
    }
};
```
