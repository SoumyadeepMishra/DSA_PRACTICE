# Evaluate Division

```cpp

class Solution {
public:

    void dfs(unordered_map<string, vector<pair<string, double>>> &adj, string src, string dst, unordered_set<string> &visited, double prod, double &ans) {

        if(ans != -1.0) 
            return; // answer already found in other branch

        if(visited.find(src) != visited.end())
            return;

        visited.insert(src);

        // check it is destination or not 
        if(src == dst) {
            ans = prod;
            return;
        }

        for(auto &p : adj[src]) {
            string v = p.first;
            double val = p.second;

            if(visited.find(v) == visited.end()) {
                dfs(adj, v, dst, visited, prod*val, ans);
            }
        }

    }

    vector<double> calcEquation(vector<vector<string>>& equations, vector<double>& values, vector<vector<string>>& queries) {
        int n = equations.size();

        // make adjacency list
        unordered_map<string, vector<pair<string, double>>> adj;
        for(int i=0;i<n;i++) {
            string u = equations[i][0];
            string v = equations[i][1];
            double val = values[i];

            adj[u].push_back({v, val});
            adj[v].push_back({u, 1.0/val});
        }

        vector<double> result;

        for(auto &query : queries) {
            string src = query[0];
            string dst = query[1];

            double ans = -1.0;
            double prod = 1.0;

            if(adj.find(src) != adj.end() && adj.find(dst) != adj.end()) {

                // take unordered set for visited tracking
                unordered_set<string> visited;
                dfs(adj, src, dst, visited, prod, ans);

            }

            result.push_back(ans);
        }

        return result;
    }
};
```
