# All Paths From Source to Target

```cpp

class Solution {
public:

    void dfs(vector<vector<int>> &graph, int u, int d, vector<vector<int>> &result, vector<int> &temp) {
        temp.push_back(u);

        // check if u == target
        if(u == d) {
            result.push_back(temp);
        }
        else {
            for(int &v : graph[u]) {
                dfs(graph, v, d, result, temp);
            }
        }
        temp.pop_back();
    }

    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        int n = graph.size();
        int source = 0;
        int target = n-1;

        vector<vector<int>> result;
        vector<int> temp;

        dfs(graph, source, target, result, temp);

        return result;
    }
};
```
