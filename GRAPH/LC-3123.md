# Find Edges in Shortest Paths

```cpp

class Solution {
public:
       
    typedef pair<int, int> p;
    typedef long long ll;

    vector<int> dijkstra(unordered_map<int, vector<p>> &adj, int src, int n) {
        vector<int> result(n, INT_MAX);
        priority_queue<p, vector<p>, greater<p> > que;
        que.push({0, src});
        result[src] = 0;

        while(!que.empty()) {

            int node = que.top().second;
            int dist = que.top().first;
            que.pop();

            for(auto &ngh : adj[node]) {
                int v = ngh.first;
                int w = ngh.second;

                if(result[v] > dist + w) {
                    result[v] = dist + w;
                    que.push({dist + w, v});
                }

            }

        }

        return result;

    }

    vector<bool> findAnswer(int n, vector<vector<int>>& edges) {
        // makae the graph
        unordered_map<int, vector<p>> adj;
        for(auto &edge : edges) {

            int u = edge[0];
            int v = edge[1];
            int w = edge[2];

            adj[u].push_back({v, w});
            adj[v].push_back({u, w});

        }

        vector<int> fromSrc = dijkstra(adj, 0, n);
        vector<int> fromDest = dijkstra(adj, n-1, n);
        vector<bool> result(edges.size(), false);

        for(int i=0;i<edges.size();i++) {

            int u = edges[i][0];
            int v = edges[i][1];
            int w = edges[i][2];

            ll distFromSrc = fromSrc[u];
            ll distFromDest = fromDest[v];

            if(distFromSrc + w + distFromDest == fromSrc[n-1]) {
                result[i] = true;
            }

            distFromSrc = fromSrc[v];
            distFromDest = fromDest[u];

            if(distFromSrc + w + distFromDest == fromSrc[n-1]) {
                result[i] = true;
            }

        }

        return result;
    }
};
```
