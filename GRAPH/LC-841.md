# Keys and Rooms

# dfs

```cpp

class Solution {
public:

    void dfs(vector<vector<int>> &rooms, int source, vector<bool> &visited) {
        // base case 
        if(visited[source] == true)
            return;

        visited[source] = true;

        // check for neighbours
        for(auto &v : rooms[source]) {
            if(visited[v] == false) {
                dfs(rooms, v, visited);
            }
        }
        return;
    }

    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        // rooms is given as a form of adjacency list
        int n = rooms.size();
        int startNode = 0;

        vector<bool> visited(n, false);
        dfs(rooms, startNode, visited);

        // check the visited array after dfs call
        for(bool visit : visited) {
            if(visit == false)
                return false;
        }
        return true;
    }
};
```

# bfs

```cpp

class Solution {
public:
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        // bfs lagiye korbo
        int n = rooms.size();
        vector<bool> visited(n, false);
        queue<int> que;
        que.push(0);
        visited[0] = true;

        while(!que.empty()) {
            int curr = que.front();
            que.pop();

            for(auto &v : rooms[curr]) {
                if(visited[v] == false) {
                    visited[v] = true;
                    que.push(v);
                }
            }
        }

        // check for any false value after bfs
        for(bool v : visited) {
            if(v == false)
                return false;
        }
        return true;
    }
};
```
