# dfs

```cpp

class Solution {
public:

    void dfs(vector<vector<int>> &rooms, int source, vector<bool> &visited) {
        // base case 
        if(visited[source] == true)
            return;

        visited[source] = true;

        // check for neighbours
        for(auto &v : rooms[source]) {
            if(visited[v] == false) {
                dfs(rooms, v, visited);
            }
        }
        return;
    }

    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        // rooms is given as a form of adjacency list
        int n = rooms.size();
        int startNode = 0;

        vector<bool> visited(n, false);
        dfs(rooms, startNode, visited);

        // check the visited array after dfs call
        for(bool visit : visited) {
            if(visit == false)
                return false;
        }
        return true;
    }
};
```
