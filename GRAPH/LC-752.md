# Open the Lock

```cpp

class Solution {
public:

    void fillNeighbours(queue<string> &que, string &curr, unordered_set<string> &st) {

        for(int i=0;i<4;i++) {
            char ch = curr[i];

            char inc = ch == '9' ? '0' : ch+1;
            char dec = ch == '0' ? '9' : ch-1;

            curr[i] = inc;

            // check it is in the deadends
            if(st.count(curr) == false) {
                que.push(curr);
                st.insert(curr);
            }

            curr[i] = dec;
            if(st.count(curr) == false) {
                que.push(curr);
                st.insert(curr);
            }

            curr[i] = ch;

        }

    }

    int openLock(vector<string>& deadends, string target) {
        int n = deadends.size();

        // set to track the deadends
        unordered_set<string> st(deadends.begin(), deadends.end());

        // start jodi deadends eage thekei thake
        if(st.count("0000"))
            return -1;

        queue<string> que;
        que.push("0000");
        st.insert("0000");
        int level = 0;

        while(!que.empty()) {

            int n = que.size();

            while(n--) {
                string curr = que.front();
                que.pop();

                if(curr == target) 
                    return level;

                fillNeighbours(que, curr, st);
            }

            level++;

        }

        return -1;
    }
};
```
