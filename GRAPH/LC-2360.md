# Longest Cycle in a Graph

```cpp

class Solution {
public:

    void dfs(int u, vector<int> &edges, vector<bool> &visited, vector<int> &count, vector<bool> &inRecursion, int &result) {

        visited[u] = true;
        inRecursion[u] = true;

        // neighbour
        int v = edges[u];

        if(v != -1) {

            if(!visited[v]) {
                count[v] = count[u] + 1;
                dfs(v, edges, visited, count, inRecursion, result);
            }
            else if(inRecursion[v]) { // cycle detected
                int cycleLength = count[u] - count[v] + 1;
                result = max(result, cycleLength);
            }
        }

        inRecursion[u] = false;
    }

    int longestCycle(vector<int>& edges) {
        int n = edges.size();

        vector<bool> visited(n, false);
        vector<int> count(n, 1); // atmost one outdegree(outgoing edge)
        vector<bool> inRecursion(n, false); // to track cycle in DG

        int result = -1;

        for(int i=0;i<n;i++) {

            if(!visited[i]) 
                dfs(i, edges, visited, count, inRecursion, result);
        }

        return result;
    }
};
```
