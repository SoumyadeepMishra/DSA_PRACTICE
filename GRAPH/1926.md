# Nearest Exit from Entrance in Maze

```cpp

class Solution {
public:

    vector<vector<int>> dirs = {{0,1}, {1,0}, {0, -1}, {-1, 0}};

    bool isSafe(int i, int j, int m, int n) {
        if(i >= 0 && j >= 0 && i < m && j < n)
            return true;

        return false;
    }

    int nearestExit(vector<vector<char>>& maze, vector<int>& entrance) {
        int m = maze.size();
        int n = maze[0].size();

        int row = entrance[0];
        int col = entrance[1];

        queue<pair<int, int>> que;
        que.push({row, col});
        maze[row][col] = '+'; // mark visited

        int steps = 0;

        // same bfs template
        while(!que.empty()) {
            int size = que.size();

            while(size--) {
                auto curr = que.front();
                que.pop();

                // check if it is the boundary node or not 
                int i = curr.first;
                int j = curr.second;

                if(!(i == row && j == col)) // means it is not the entrance
                    if(i == 0 || j == 0 || i == m-1 || j == n-1) 
                        return steps;

                // check for neighbours
                for(auto &dir : dirs) {
                    int i_ = i + dir[0];
                    int j_ = j + dir[1];

                    // if safe or not and if not visited
                    if(isSafe(i_,j_,m,n) && maze[i_][j_] != '+') {
                        que.push({i_,j_});
                        maze[i_][j_] = '+';
                    }
                }

            }
            steps++;
        }
        return -1;
    }
};
```
