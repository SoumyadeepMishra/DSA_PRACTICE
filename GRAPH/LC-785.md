#  Is Graph Bipartite?

# BFS

```cpp

class Solution {
public:

    bool bfs(vector<vector<int>> &adj, int u, vector<int> &color, int curr_color) {
        color[u] = curr_color;
        queue<int> que;
        que.push(u);

        while(!que.empty()) {
            int u = que.front();
            que.pop();

            for(int &v : adj[u]) {
                if(color[v] == color[u]) return false;

                else if(color[v] == -1) {
                    color[v] = 1 - color[u];
                    que.push(v);
                }
            }
        }
        return true;
    }

    bool isBipartite(vector<vector<int>>& graph) {
        // will solve using bfs
        vector<int> color(graph.size(), -1);

        for(int i=0;i<graph.size();i++) {
            if(color[i] == -1) {
                if(bfs(graph, i, color, 1) ==  false) return false;
            }
        }

        return true;
    }
};
```

# DFS

```CPP

class Solution {
public:

    bool dfs(vector<vector<int>> &adj, int u, vector<int> &color, int currColor) {
        color[u] = currColor;

        // adjacent nodes
        for(int &v : adj[u]) {
            if(color[v] == color[u]) return false;

            if(color[v] == -1) {
                if(dfs(adj, v, color, 1-currColor) == false) return false;
            }
        }

        return true;
    }

    bool isBipartite(vector<vector<int>>& graph) {
        int n = graph.size();

        vector<int> color(n, -1);

        for(int i=0;i<n;i++) {
            if(color[i] == -1) {
                if(dfs(graph, i, color, 1) == false) return false;
            }
        }

        return true;
    }
};
```
