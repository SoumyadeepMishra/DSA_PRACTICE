# Find Closest Node to Given Two Nodes

# DFS

```cpp

class Solution {
public:

    void dfs(vector<int> &edges, int source, vector<int> &dist, vector<bool> &visited) {
        visited[source] = true;

        // as only one neighbour
        int v = edges[source];

        if(v != -1 && visited[v] == false) {
            visited[v] = true;
            dist[v] = 1 + dist[source];
            dfs(edges, v, dist, visited);
        } 

    }

    int closestMeetingNode(vector<int>& edges, int node1, int node2) {
        int n = edges.size();

        // dist of each node from node1 and node2
        vector<int> dist1(n, INT_MAX);
        vector<int> dist2(n, INT_MAX);

        vector<bool> visited1(n, false);
        vector<bool> visited2(n, false);

        // call dfs to compute the distance array for two nodes
        dist1[node1] = 0;
        dist2[node2] = 0;

        dfs(edges, node1, dist1, visited1);
        dfs(edges, node2, dist2, visited2);

        int minDist = INT_MAX;
        int minDistNode = -1;

        for(int i=0;i<n;i++) {
            int maxD = max(dist1[i], dist2[i]);

            if(maxD < minDist) {
                minDist = maxD;
                minDistNode = i;
            }
        }
        return minDistNode;
    }
};

``

# BFS

```cpp

class Solution {
public:

    void bfs(vector<int> &edges, int source, vector<int> &dist) {
        // simple bfs structure
        vector<bool> visited(edges.size(), false);
        queue<int> que;
        que.push(source);
        visited[source] = true;
        dist[source] = 0;

        while(!que.empty()) {
            int u = que.front();
            que.pop();

            int v = edges[u];
            if(v != -1 && !visited[v]) {
                visited[v] = true;
                dist[v] = 1 + dist[u];
                que.push(v);
            }
        }
    }

    int closestMeetingNode(vector<int>& edges, int node1, int node2) {
        int n = edges.size();

        // dist of each node from node1 and node2
        vector<int> dist1(n, INT_MAX);
        vector<int> dist2(n, INT_MAX);

        bfs(edges, node2, dist2);
        bfs(edges, node1, dist1);

        int minDist = INT_MAX;
        int minDistNode = -1;

        for(int i=0;i<n;i++) {
            int maxD = max(dist1[i], dist2[i]);

            if(maxD < minDist) {
                minDist = maxD;
                minDistNode = i;
            }
        }
        return minDistNode;
    }
};
