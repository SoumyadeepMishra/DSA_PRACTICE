# Shortest Path in a Grid with Obstacles Elimination

```cpp

class Solution {
public:
    vector<vector<int>> directions = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();

        // for bfs take queue
        queue<vector<int>> que;  // {x_coordinate, y_coordinate, obstacle}
        que.push({0, 0, k});
        
        // unordered_set also can be used here set<vector<int>> st;
        bool visited[41][41][1601];
        memset(visited, false, sizeof(visited));
        visited[0][0][k] = true;
        int steps = 0;

        while(!que.empty()) {

            int size = que.size();
            while(size--) {

                auto temp = que.front();
                que.pop();
                int curr_x = temp[0];
                int curr_y = temp[1];
                int obstacle = temp[2];

                if(curr_x == m-1 && curr_y == n-1) {
                    return steps;
                }

                for(auto &dir : directions) {
                    int new_x = curr_x + dir[0];
                    int new_y = curr_y + dir[1];
                    
                    // checking the validity
                    if(new_x < 0 || new_x >= m || new_y < 0 || new_y >= n)
                        continue;

                    if(grid[new_x][new_y] == 0 && !visited[new_x][new_y][obstacle]) {
                        que.push({new_x, new_y, obstacle});
                        visited[new_x][new_y][obstacle] = true;
                    }
                    else if(grid[new_x][new_y] == 1 && obstacle > 0 && !visited[new_x][new_y][obstacle-1]) {
                        que.push({new_x, new_y, obstacle-1});
                        visited[new_x][new_y][obstacle-1] = true;
                    }
                }

            }

            steps++;

        }

        return -1;
    }
};
```
