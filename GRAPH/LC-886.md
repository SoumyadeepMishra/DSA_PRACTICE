# Possible Bipartition

```cpp

class Solution {
public:

    bool checkBipartite(unordered_map<int, vector<int>> &adj, int node, vector<int> &color) {
        // declare a queue
        queue<int> que;
        que.push(node);
        color[node] = 1; // ekhane 1 mane RED babohar korlam rrrr 0 mane GREEN 

        while(!que.empty()) {
            int curr = que.front();
            que.pop();

            // main conditions here
            for(auto &v : adj[curr]) {
                // color same as parent
                if(color[v] == color[curr])
                    return false;

                if(color[v] == -1) {
                    que.push(v);
                    color[v] = 1 - color[curr];
                }
            }
        }

        return true;
    }

    bool possibleBipartition(int n, vector<vector<int>>& dislikes) {
        // n is the number of nodes
        // first build the adjacency list
        unordered_map<int, vector<int> > adj;
        for(auto &dis : dislikes) {
            int u = dis[0];
            int v = dis[1];

            adj[u].push_back(v);
            adj[v].push_back(u);
        }

        // color vector jeta visited vector ero kaj korbe 
        vector<int> color(n+1, -1);

        // check for disconnected graph
        for(int i=1;i<+n;i++) {
            if(color[i] == -1) {
                if(checkBipartite(adj, i, color) == false)
                    return false;
            }
        }
        return true;
    }
};
```
