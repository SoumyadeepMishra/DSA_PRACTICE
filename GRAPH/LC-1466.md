# Reorder Routes to Make All Paths Lead to the City Zero

```cpp

class Solution {
public:

    void dfs(unordered_map<int, vector<pair<int, int>>> &adj, int u, int parent, int &count) {
        
        for(auto &p : adj[u]) {
            int v = p.first;
            int asolNokol = p.second;

            if(v != parent) {

                if(asolNokol == 1) // real edge jeta 0 theke dure jacche
                    count++;
                
                dfs(adj, v, u, count);
            }
        }
    }

    int minReorder(int n, vector<vector<int>>& connections) {
        // first make the adjacency list
        // make alternative edge also
        unordered_map<int, vector<pair<int, int>>> adj;
        for(auto &c : connections) {
            int u = c[0];
            int v = c[1];

            adj[u].push_back({v, 1}); // 1-->asol edge
            adj[v].push_back({u, 0});
        }
        
        int count = 0;
        dfs(adj, 0, -1, count);

        return count;
    }
};
```
