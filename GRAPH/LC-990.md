# Satisfiability of Equality Equations

```cpp

class Solution {
public:

    vector<int> parent;
    vector<int> rank;

    int find(int x) {
        if(x == parent[x])
            return x;
        
        return parent[x] = find(parent[x]);
    }

    void union_rank(int x, int y) {
        int x_parent = find(x);
        int y_parent = find(y);

        if(x_parent == y_parent) return;

        else if(rank[x_parent] > rank[y_parent]) {
            parent[y_parent] = x_parent;
        }
        else if(rank[x_parent] < rank[y_parent]) {
            parent[x_parent] = y_parent;
        }
        else {
            parent[y_parent] = x_parent;
            rank[x_parent]++;
        }
    }

    bool equationsPossible(vector<string>& equations) {
        // will soolve it using union find algorithm 
        parent.resize(26);
        rank.resize(26, 0);

        for(int i=0;i<26;i++) {
            parent[i] = i;
        }

        // search for == part and do union to make a graph
        for(string &s : equations) {
            if(s[1] == '=') {
                union_rank(s[0]-'a', s[3]-'a');
            }
        }

        // search for != to check validity
        for(string &s : equations) {
            if(s[1] == '!') {
                int p1 = find(s[0]-'a');
                int p2 = find(s[3]-'a');

                if(p1 == p2) return false;
                
            }
        }

        return true;

    }
};
```
