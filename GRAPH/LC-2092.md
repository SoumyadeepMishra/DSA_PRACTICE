# Find All People With Secret

```cpp

class Solution {
public:
    typedef pair<int, int> p;
    vector<int> findAllPeople(int n, vector<vector<int>>& meetings, int firstPerson) {
        map<int, vector<p>> timeMeetings;

        for(auto &meeting : meetings) {
            int p1 = meeting[0];
            int p2 = meeting[1];
            int time = meeting[2];

            timeMeetings[time].push_back({p1, p2});
        }

        vector<bool> knowsSecret(n, false);
        knowsSecret[0] = true;
        knowsSecret[firstPerson] = true;

        // traverse in the map
        for(auto &it : timeMeetings) {

            int time = it.first;
            auto p = it.second;

            unordered_map<int, vector<int>> adj;
            queue<int> que;
            unordered_set<int> alreadyAdded;

            for(auto &v : p) {
                int person1  = v.first;
                int person2 = v.second;

                adj[person1].push_back(person2);
                adj[person2].push_back(person1);

                if(knowsSecret[person1] == true && alreadyAdded.count(person1) == false) {
                   que.push(person1);
                   alreadyAdded.insert(person1);
                }

                if(knowsSecret[person2] == true && alreadyAdded.count(person2) == false) {
                    que.push(person2);
                    alreadyAdded.insert(person2);
                }
            }

            while(!que.empty()) {
                int person  = que.front();
                que.pop();

                for(int &v : adj[person]) {
                    if(knowsSecret[v] == false) {
                        knowsSecret[v] = true;
                        que.push(v);
                    }
                }
            }

        }

        vector<int> result;
        for(int i=0;i<n;i++) {
            if(knowsSecret[i] == true) {
                result.push_back(i);
            }
        }

        return result;
    }
};
```
