# Minimum Score of a Path Between Two Cities

```cpp

class Solution {
public:

    void dfs(unordered_map<int, vector<pair<int, int>>> &adj, int u, vector<bool> &visited, int &result) {
        visited[u] = true;

        // visit neighbours
        for(auto &p : adj[u]) {
            int v = p.first;
            int cost = p.second;

            result = min(result, cost);

            if(!visited[v]) 
                dfs(adj, v, visited, result);
        }
        
    }

    int minScore(int n, vector<vector<int>>& roads) {
        // becomes simple traversal problem
        // score = min(costs of each path)

        unordered_map<int, vector<pair<int, int>>> adj;
        for(auto &road : roads) {
            int u = road[0];
            int v = road[1];
            int wt = road[2];

            adj[u].push_back({v, wt});
            adj[v].push_back({u, wt});
        }

        vector<bool> visited(n, false);
        int result = INT_MAX;

        dfs(adj, 1, visited, result);

        return result;
    }
};
```

# BFS

```cpp

class Solution {
public:
    int minScore(int n, vector<vector<int>>& roads) {
        unordered_map<int, vector<pair<int, int>>> adj;
        for(auto &road : roads) {
            int u = road[0];
            int v = road[1];
            int wt = road[2];

            adj[u].push_back({v, wt});
            adj[v].push_back({u, wt});
        }

        vector<bool> visited(n, false);
        int result = INT_MAX;

        // bfs
        queue<int> que;
        que.push(1); // starting node
        visited[1] = true;

        while(!que.empty()) {
            int u = que.front();
            que.pop();

            for(auto &p : adj[u]) {
                int v = p.first;
                int cost = p.second;

                result = min(result, cost);
                if(!visited[v]){
                    que.push(v);
                    visited[v] = true;
                }
            }
        }
        return result;
    }
};
```
