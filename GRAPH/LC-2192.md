# All Ancestors of a Node in a Directed Acyclic Graph

```cpp

class Solution {
public:

    void dfs(int ancestor, int u, unordered_map<int, vector<int>> &adj, vector<vector<int>> &result) {

        for(int &v : adj[u]) {
            if(result[v].empty() || result[v].back() != ancestor) {
                result[v].push_back(ancestor);
                dfs(ancestor, v, adj, result);
            }
        }

    }

    vector<vector<int>> getAncestors(int n, vector<vector<int>>& edges) {
        // make graph
        unordered_map<int, vector<int>> adj;
        for(auto &edge : edges) {

            int u = edge[0];
            int v = edge[1];
            adj[u].push_back(v);

        }

        vector<vector<int>> result(n);

        // traverse through dfs and check for each node as ancestor
        for(int i=0;i<n;i++) {

            int ancestor = i;
            dfs(ancestor, i, adj, result);

        }

        return result;
    }
};
```
