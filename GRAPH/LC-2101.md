# Detonate the Maximum Bombs

```cpp

class Solution {
public:

    void dfs(int u, unordered_map<int, vector<int>> &adj, unordered_set<int> &visited) {

        visited.insert(u);

        for(int &v : adj[u]) {

            if(visited.find(v) == visited.end())
                dfs(v, adj, visited);

        }

    }

    int maximumDetonation(vector<vector<int>>& bombs) {
        int n = bombs.size();
        
        // make adjacency list
        unordered_map<int, vector<int>> adj;
        for(int i=0;i<n;i++) {
            
                long long x1 = bombs[i][0];
                long long y1 = bombs[i][1];
                long long r1 = bombs[i][2];

            for(int j=0;j<n;j++) {

                if(i == j) 
                    continue;

                long long x2 = bombs[j][0];
                long long y2 = bombs[j][1];
                long long r2 = bombs[j][2];


                long long dist = (x2-x1)*(x2-x1) + (y2-y1)*(y2-y1);

                if((long long) r1*r1 >= dist)
                    adj[i].push_back(j);

            }
        }

        int result = 0;
        for(int i=0;i<n;i++) {

            unordered_set<int> visited;

            dfs(i, adj, visited);
            int maxNodes = visited.size();
            result = max(result, maxNodes);

        }

        return result;
    }
};
```
