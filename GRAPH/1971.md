#  Find if Path Exists in Graph

```cpp

class Solution {
public:

    bool checkPath(unordered_map<int, vector<int>> &mp, int s, int d, vector<bool> &visited) {
        // min true condition is the base condition
        if(s == d)
            return true;
        
        visited[s] = true;

        for(auto &v : mp[s]) {
            if(!visited[v]) {
                if(checkPath(mp, v, d, visited)) 
                    return true;
            }
        }
        return false;
    }

    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        // build the adjacency list of the graph
        unordered_map<int, vector<int> > mp;
        for(auto &edge : edges) {
            int u = edge[0];
            int v = edge[1];

            mp[u].push_back(v);
            mp[v].push_back(u);
        }

        vector<bool> visited(n, false);

        return checkPath(mp, source, destination, visited);
    }
};

```
