# Minimum Genetic Mutation

```cpp

class Solution {
public:
    int minMutation(string startGene, string endGene, vector<string>& bank) {
        // vector ke set e transfer korbo jate access korte subidha hoy
        unordered_set<string> bankSet(bank.begin(), bank.end());
        unordered_set<string> visited;
        // apply bfs
        queue<string> que;
        que.push(startGene);
        visited.insert(startGene);
        int level = 0; // steps

        while(!que.empty()) {
            int n = que.size();
            while(n--) {
                string curr = que.front();
                que.pop();

                // check it is the end or not 
                if(curr == endGene)
                    return level;

                // then check for each every combination
                for(auto ch : "ACGT") {
                    for(int i=0;i<curr.length();i++) {
                        string ngh = curr;
                        ngh[i] = ch;

                        // check the new string present in bank and visited
                        if(visited.find(ngh) == visited.end() && bankSet.find(ngh) == bankSet.end()) {
                            que.push(ngh);
                            visited.insert(ngh);
                        }
                    }
                }
            }
            level += 1;
        }
        return -1;
    }
};

```

# optimal --> remove seen element directly from the bank no need of visited set

```cpp

class Solution {
public:
    int minMutation(string startGene, string endGene, vector<string>& bank) {
        // vector ke set e transfer korbo jate access korte subidha hoy
        unordered_set<string> bankSet(bank.begin(), bank.end());
        
        // apply bfs
        queue<string> que;
        que.push(startGene);
        
        int level = 0; // steps

        while(!que.empty()) {
            int n = que.size();
            while(n--) {
                string curr = que.front();
                que.pop();

                // check it is the end or not 
                if(curr == endGene)
                    return level;

                // then check for each every combination
                for(auto ch : "ACGT") {
                    for(int i=0;i<curr.length();i++) {
                        string ngh = curr;
                        ngh[i] = ch;

                        // check the new string present in bank and visited
                        if(bankSet.find(ngh) != bankSet.end()) {
                            que.push(ngh);
                            bankSet.erase(ngh);
                        }
                    }
                }
            }
            level += 1;
        }
        return -1;
    }
};
```
