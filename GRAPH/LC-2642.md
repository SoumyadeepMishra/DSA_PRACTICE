# Design Graph With Shortest Path Calculator

```cpp

class Graph {
public:
    vector<vector<int>> grid;
    int N;
    Graph(int n, vector<vector<int>>& edges) {
        N = n;
        grid = vector<vector<int>>(n, vector<int>(n, 1e9));

        for(auto &edge : edges) {
            int u = edge[0];
            int v = edge[1];
            int cost = edge[2];

            grid[u][v] = cost;
        }

        // diagonal element should be zero.... as u --> u cost should be zero
        for(int i=0;i<n;i++) {
            grid[i][i] = 0;
        }

        // floyd warshall algorithm
        for(int k=0;k<n;k++) {
            for(int i=0;i<n;i++) {
                for(int j=0;j<n;j++) 
                    grid[i][j] = min(grid[i][j], grid[i][k] + grid[k][j]);
                
            }
        }
    }
    
    void addEdge(vector<int> edge) {
        int u = edge[0];
        int v = edge[1];
        int cost = edge[2];

        // update matrix for this edge only 
        for(int i=0;i<N;i++) {
            for(int j=0;j<N;j++) {
                grid[i][j] = min(grid[i][j], grid[i][u] + cost + grid[v][j]);
            }
        }
    }
    
    int shortestPath(int node1, int node2) {
        return grid[node1][node2] == 1e9 ? -1 : grid[node1][node2];
    }
};

/**
 * Your Graph object will be instantiated and called as such:
 * Graph* obj = new Graph(n, edges);
 * obj->addEdge(edge);
 * int param_2 = obj->shortestPath(node1,node2);
```
