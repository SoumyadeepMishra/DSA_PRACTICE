# Cheapest Flights Within K Stops

```cpp

class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int k) {
        unordered_map<int, vector<pair<int, int>>> adj;
        for(auto &flight : flights) {
            int u = flight[0];
            int v = flight[1];
            int cost = flight[2];

            adj[u].push_back({v, cost});
        }
        vector<int> distance(n, INT_MAX);

        // level order traverse korbo and steps count korbo
        int steps = 0;
        queue<pair<int, int>> que; // {node,cost}
        que.push({src, 0});
        distance[src] = 0;

        while(!que.empty() && steps <= k) {
            int size = que.size();

            while(size--) {
                int u = que.front().first;
                int cost = que.front().second; // cost to reach u
                que.pop();

                for(auto &ngh : adj[u]) {
                    int v = ngh.first;
                    int wt = ngh.second;

                    if(cost + wt < distance[v]) {
                        distance[v] = cost + wt;
                        que.push({v, cost + wt});
                    }
                }

            }
            steps++;
        }
        return distance[dst] == INT_MAX ? -1 : distance[dst];
    }
};

```

# Bellman Ford approach

```cpp

class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int k) {
        
        vector<int> dist(n, 1e9);   // initially sab infinity
        dist[src] = 0;              // source ka cost 0
        
        // K+1 times relaxation
        for(int i = 0; i <= k; i++) {
             vector<int> temp = dist;   // copy array
            
            for(auto &flight : flights) {
                int u = flight[0];
                int v = flight[1];
                int cost = flight[2];
                
                // agar u reachable hai
                if(dist[u] != 1e9 && dist[u] + cost < temp[v]) {
                    temp[v] = dist[u] + cost;
                }
            }
            
           dist = temp;   // update
        }
        
        return dist[dst] == 1e9 ? -1 : dist[dst];
    }
};
```
