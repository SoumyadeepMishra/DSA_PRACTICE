# Find Eventual Safe States

```cpp

class Solution {
public:

    bool checkCycle(int u, vector<vector<int>> &adj, vector<bool> &visited, vector<bool> &inRecursion) {

        visited[u] = true;
        inRecursion[u] = true;

        for(int &v : adj[u]) {

            if(!visited[v]) {
                if(checkCycle(v, adj, visited, inRecursion))
                    return true;
            }
            else if(inRecursion[v] == true) 
                return true;

        }

        inRecursion[u] = false;
        return false;

    }

    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int n = graph.size();

        vector<bool> visited(n, false);
        vector<bool> inRecursion(n, false);

        for(int i=0;i<n;i++) {
            if(!visited[i]) {
                checkCycle(i, graph, visited, inRecursion);
            }
        }

        vector<int> ans;
        for(int i=0;i<n;i++) {

            if(inRecursion[i] == false)
                ans.push_back(i);

        }

        return ans;
    }

};
```
