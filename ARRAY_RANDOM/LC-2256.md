# Minimum Average Difference

```cpp

class Solution {
public:
    int minimumAverageDifference(vector<int>& nums) {
        int n = nums.size();

        long long totalSum = 0;
        for(int i=0;i<n;i++) {
            totalSum += nums[i];
        }

        int result = INT_MAX;
        int idx = -1;
        long long leftSum = 0;
        long long rightSum = 0;

        for(int i=0;i<n;i++) {

            leftSum += nums[i];
            int n1 = i + 1;
            rightSum = totalSum - leftSum;
            int n2 = n - n1;

            int leftAvg = leftSum / n1;
            int rightAvg = 0;
            if(i == n-1) {
                rightAvg = 0;
            }
            else {
                rightAvg = rightSum / n2;
            }

            int diff = abs(leftAvg - rightAvg);
            if(diff < result) {
                result = diff;
                idx = i;
            }

        }

        return idx;
    }
};
```
