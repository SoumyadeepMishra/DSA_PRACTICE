# Make Sum Divisible by P

```cpp

class Solution {
public:
    int minSubarray(vector<int>& nums, int p) {
        int n = nums.size();

        long long sum = 0;
        for(int &x : nums) {
            sum += x;
        }

        int target = sum % p;
        if(target == 0) {
            return 0;
        }

        int result = INT_MAX;
        int currSum = 0;
        unordered_map<int, int> mp;
        mp[0] = -1;

        for(int j=0;j<n;j++) {

            currSum = (currSum + nums[j]) % p;
            int prev = (currSum - target + p) % p;

            if(mp.count(prev) == true) {
                result = min(result, (j-mp[prev]));
            }
            mp[currSum] = j;

        }

        return (result == INT_MAX || result == n) ? -1 : result;
        
    }
};
```
