# Where Will the Ball Fall

```cpp

class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> result; // number of balls

        // for every ball
        for(int i=0;i<n;i++) {
            
            int row = 0;
            int col = i;
            bool atka = false;

            while(row < m && col < n) {
                if(grid[row][col] == 1) {
                    if(col == n-1 || grid[row][col+1] == -1) {
                        atka = true;
                        break;
                    }
                    row++;
                    col++;
                }
                else if(grid[row][col] == -1) {
                    if(col == 0 || grid[row][col-1] == 1) {
                        atka = true;
                        break;
                    }
                    row++;
                    col--;
                }
            }

            if(atka == true) {
                result.push_back(-1);
            }
            else {
                result.push_back(col);
            }

        }

        return result;
    }
};
```
