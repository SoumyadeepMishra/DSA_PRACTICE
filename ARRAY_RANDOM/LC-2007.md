# Find Original Array From Doubled Array

```cpp

class Solution {
public:
    vector<int> findOriginalArray(vector<int>& changed) {
        int n = changed.size();
        if(n%2 != 0) {
            return {};
        }
        sort(begin(changed), end(changed));

        unordered_map<int, int> mp;
        // store frequency in the map
        for(int &num : changed) {
            mp[num]++;
        }

        vector<int> result;

        // traverse and check if double exists
        for(int &num : changed) {

            if(mp[num] == 0)
                continue;

            int twice = num * 2;
            if(!mp.count(twice) || mp[twice] == 0) {
                return {};
            }
            result.push_back(num);
            mp[num]--;
            mp[twice]--;

        }

        return result;
    }
};
```
