# Subarray Sums Divisible by K

```cpp

class Solution {
public:
    int subarraysDivByK(vector<int>& nums, int k) {
        int n = nums.size();
        unordered_map<int, int> mp;
        mp[0] = 1;  // {remainder, count}

        int count = 0;
        int sum = 0;

        for(int i=0;i<n;i++) {

            sum += nums[i];
            
            int remainder = sum % k;
            if(remainder < 0) {
                remainder += k;
            }

            if(mp.count(remainder) == true) {
                count += mp[remainder];
            }
            mp[remainder]++;

        }

        return count;
    }
};
```
