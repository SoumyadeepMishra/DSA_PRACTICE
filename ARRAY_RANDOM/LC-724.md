# Find Pivot Index

```cpp

class Solution {
public:
    int pivotIndex(vector<int>& nums) {
        int n = nums.size();

        int sum = 0;
        for(int &x : nums) {
            sum += x;
        }

        int currSum = 0;
        for(int i=0;i<n;i++) {

            int ls = currSum;
            int rs = sum - currSum - nums[i];

            if(ls == rs) 
                return i;

            currSum += nums[i];

        }

        return -1;
    }
};
```
