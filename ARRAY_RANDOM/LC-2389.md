# Longest Subsequence With Limited Sum

```cpp

class Solution {
public:
    int binarySearch(vector<int> &nums, int n, int target) {
        int left = 0;
        int right = n-1;
        int resultIdx = -1; 

        while(left <= right) {

            int mid = (left + (right - left)/2);

            if(nums[mid] <= target) {
                resultIdx = mid;
                left = mid + 1;
            }
            else {
                right = mid - 1;
            }
        
        }

        return resultIdx + 1;
    }


    vector<int> answerQueries(vector<int>& nums, vector<int>& queries) {
        int n = nums.size();
        sort(begin(nums), end(nums));

        // total prefix sum
        for(int i=1;i<n;i++) {
            nums[i] += nums[i-1];
        }

        vector<int> result;
        int ans = -1;
        
        for(int &query : queries) {

            ans = binarySearch(nums, n, query);
            result.push_back(ans);
        }

        return result;
    }
};
```
